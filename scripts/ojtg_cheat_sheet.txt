# 任意のコンテストの問題全てに対して、解答テンプレートファイルやランダムテスト生成コードを含むフォルダ作成
oj-prepare https://atcoder.jp/contests/abc177

# ランダムテストケースを生成(以下の例だと10件)　※対象の問題フォルダ内で実行する必要あり
oj g/i "python generate.py" 10

# ランダムテストケースに対する出力を生成　※対象の問題フォルダ内で実行する必要あり
oj g/o -c "python ../a.py"

# 任意の問題のサンプルテストケースを実行　※対象の問題フォルダ内で実行する必要あり　※testはtでも可
oj test -c "python ../a.py"

# 解答コードの提出　※対象の問題フォルダ内で実行する必要あり　※submitはsでも可
oj submit ../a.py
